#include<iostream>
using namespace std;
int prime(int n)
{
	bool IsPrime = true;
	for (int i = 2; i <= n / 2; i++)
	{
		if (n % i == 0)
		{
			IsPrime = false;
			break;
			cout << "NOT PRIME" << endl;
		}
		
	}
	if (IsPrime == true)
	{
		return 0;
		cout << "PRIME" << endl;
	}
	else
	{
		return -1;
	}
}
